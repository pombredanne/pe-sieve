19 virtual bool processThunks(LPSTR lib_name, ULONG_PTR origFirstThunkPtr, ULONG_PTR firstThunkPtr)
20 {
21 if (this->is64b) {
22 IMAGE_THUNK_DATA64* desc = reinterpret_cast<IMAGE_THUNK_DATA64*>(origFirstThunkPtr);
23 ULONGLONG* call_via = reinterpret_cast<ULONGLONG*>(firstThunkPtr);
24 return processThunks_tpl<ULONGLONG, IMAGE_THUNK_DATA64>(lib_name, desc, call_via, IMAGE_ORDINAL_FLAG64);
25 }
26 IMAGE_THUNK_DATA32* desc = reinterpret_cast<IMAGE_THUNK_DATA32*>(origFirstThunkPtr);
27 DWORD* call_via = reinterpret_cast<DWORD*>(firstThunkPtr);
28 return processThunks_tpl<DWORD, IMAGE_THUNK_DATA32>(lib_name, desc, call_via, IMAGE_ORDINAL_FLAG32);
29 }

32 template <typename T_FIELD, typename T_IMAGE_THUNK_DATA>
33 bool processThunks_tpl(LPSTR lib_name, T_IMAGE_THUNK_DATA* desc, T_FIELD* call_via, T_FIELD ordinal_flag)
34 {
35 ULONGLONG call_via_rva = ((ULONG_PTR)call_via - (ULONG_PTR)this->modulePtr);
36 T_FIELD raw_ordinal = 0;
37 bool is_by_ord = (desc->u1.Ordinal & ordinal_flag) != 0;
38 if (is_by_ord) {
39 raw_ordinal = desc->u1.Ordinal & (~ordinal_flag);
40#ifdef _DEBUG
41 std::cout << "raw ordinal: " << std::hex << raw_ordinal << std::endl;
42#endif
43 this->storedFunc[call_via_rva] = peconv::ExportedFunc(peconv::get_dll_shortname(lib_name), raw_ordinal);
44 }
45 else {
46 PIMAGE_IMPORT_BY_NAME by_name = (PIMAGE_IMPORT_BY_NAME)((ULONGLONG)modulePtr + desc->u1.AddressOfData);
47 LPSTR func_name = reinterpret_cast<LPSTR>(by_name->Name);
48 raw_ordinal = by_name->Hint;
49 this->storedFunc[call_via_rva] = peconv::ExportedFunc(peconv::get_dll_shortname(lib_name), func_name, raw_ordinal);
50 }
51 return true;
52 }

60const bool IATScanReport::hooksToJSON(std::stringstream &outs, size_t level)
61{
62 if (notCovered.count() == 0) {
63 return false;
64 }
65 bool is_first = true;
66 OUT_PADDED(outs, level, "\"hooks_list\" : [\n");

85 OUT_PADDED(outs, (level + 1), "\"func_name\" : ");
86 outs << "\"" << func.toString() << "\"" << ",\n";
87 }
88
89 OUT_PADDED(outs, (level + 1), "\"target\" : ");
90 outs << "\"" << std::hex << (ULONGLONG)addr << "\"";
91
92 outs << "\n";
93 OUT_PADDED(outs, level, "}");
94 }

100bool IATScanReport::saveNotRecovered(IN std::string fileName,
101 IN HANDLE hProcess,
102 IN const std::map<ULONGLONG, peconv::ExportedFunc> *storedFunc,
103 IN peconv::ImpsNotCovered &notCovered,
104 IN const ModulesInfo &modulesInfo,
105 IN const peconv::ExportsMapper *exportsMap)
106{
107 const char delim = ';';
108
109 if (notCovered.count() == 0) {
110 return false;
111 }
112 std::ofstream report;
113 report.open(fileName);
114 if (report.is_open() == false) {
115 return false;
116 }
117
118 std::map<ULONGLONG,ULONGLONG>::iterator itr;
119 for (itr = notCovered.thunkToAddr.begin(); itr != notCovered.thunkToAddr.end(); ++itr)
120 {
121 const ULONGLONG thunk = itr->first;
122 const ULONGLONG addr = itr->second;
123 report << std::hex << thunk << delim;
124
125 if (storedFunc) {
126 std::map<ULONGLONG, peconv::ExportedFunc>::const_iterator found = storedFunc->find(thunk);
127 if (found != storedFunc->end()) {
128 const peconv::ExportedFunc &func = found->second;
129 report << func.toString();
130 }
131 else {
132 report << "(unknown)";
133 }
134 report << "->";
135 }
136
137 if (exportsMap) {
138 ScannedModule *modExp = modulesInfo.findModuleContaining(addr);
139 ULONGLONG module_start = (modExp) ? modExp->getStart() : peconv::fetch_alloc_base(hProcess, (BYTE*)addr);
140
141 const peconv::ExportedFunc* func = exportsMap->find_export_by_va(addr);
142 if (func) {
143 report << func->toString();
144 }
145 else {
146 if (module_start == 0) {
147 report << "(invalid)";
148 }
149 else {
150 char moduleName[MAX_PATH] = { 0 };
151 if (GetModuleBaseNameA(hProcess, (HMODULE)module_start, moduleName, sizeof(moduleName))) {
152 report << peconv::get_dll_shortname(moduleName) << ".(unknown_func)";
153 }
154 else {
155 report << "(unknown)";
156 }
157 }
158 }
159
160 size_t offset = addr - module_start;
161 report << delim << std::hex << module_start << "+" << offset;
162
163 if (modExp) {
164 report << delim << modExp->isSuspicious();
165 }
166 }
167 report << std::endl;
168 }
169 report.close();
170 return true;
171}

199IATScanReport* pesieve::IATScanner::scanRemote()
200{
201 if (!remoteModData.isInitialized()) {
202 std::cerr << "[-] Failed to initialize remote module header" << std::endl;
203 return nullptr;
204 }
205 if (!hasImportTable(remoteModData)) {
206 IATScanReport *report = new IATScanReport(processHandle, remoteModData.modBaseAddr, remoteModData.getModuleSize(), moduleData.szModName);
207 report->status = SCAN_NOT_SUSPICIOUS;
208 return report;
209 }
210 if (!remoteModData.loadFullImage()) {
211 std::cerr << "[-] Failed to initialize remote module" << std::endl;
212 return nullptr;
213 }
214 BYTE *vBuf = remoteModData.imgBuffer;
215 size_t vBufSize = remoteModData.imgBufferSize;
216 if (!vBuf) {
217 return nullptr;
218 }
219 peconv::ImpsNotCovered not_covered;
220 peconv::fix_imports(vBuf, vBufSize, exportsMap, &not_covered);
221
222 t_scan_status status = SCAN_NOT_SUSPICIOUS;
223 if (not_covered.count() > 0) {
224 status = SCAN_SUSPICIOUS;
225 }
226
227 IATScanReport *report = new IATScanReport(processHandle, remoteModData.modBaseAddr, remoteModData.getModuleSize(), moduleData.szModName);
228 if (!report) {
229 return nullptr;
230 }
231
232 if (not_covered.count()) {
233 listAllImports(report->storedFunc);
234 }
235 if (this->hooksFilter != PE_IATS_UNFILTERED) {
236 filterResults(not_covered, *report);
237 }
238 else {
239 report->notCovered = not_covered;
240 }
241 report->status = status;
242 if (report->countHooked() == 0) {
243 report->status = SCAN_NOT_SUSPICIOUS;
244 }
245 return report;

262bool pesieve::IATScanner::filterResults(peconv::ImpsNotCovered &notCovered, IATScanReport &report)
263{
264 std::map<ULONGLONG, ULONGLONG>::iterator itr;
265 for (itr = notCovered.thunkToAddr.begin(); itr != notCovered.thunkToAddr.end(); ++itr)
266 {
267 const ULONGLONG thunk = itr->first;
268 const ULONGLONG addr = itr->second;
269
270 ScannedModule *modExp = modulesInfo.findModuleContaining(addr);
271 ULONGLONG module_start = (modExp) ? modExp->getStart() : peconv::fetch_alloc_base(this->processHandle, (BYTE*)addr);
272 if (module_start == 0) {
273 // invalid address of the hook
274 report.notCovered.insert(thunk, addr);
275 continue;
276 }
277 if (this->hooksFilter == PE_IATS_CLEAN_SYS_FILTERED) {
278 // insert hooks leading to suspicious modules:
279 if (modExp && modExp->isSuspicious()) {
280 report.notCovered.insert(thunk, addr);
281 continue;
282 }
283 }
284 // filter out hooks leading to system DLLs
285 char moduleName[MAX_PATH] = { 0 };
286 if (GetModuleFileNameExA(this->processHandle, (HMODULE)module_start, moduleName, sizeof(moduleName))) {
287 std::string dirName = peconv::get_directory_name(moduleName);
288 std::transform(dirName.begin(), dirName.end(), dirName.begin(), tolower);
289#ifdef _DEBUG
290 std::cout << "Module dir name: " << dirName << "\n";
291#endif
292 if (dirName == m_system32Path_str || dirName == m_sysWow64Path_str) {
293#ifdef _DEBUG
294 std::cout << "Skipped: " << dirName << "\n";
295#endif
296 continue;
297 }
298 }
299 // insert hooks leading to non-system modules:
300 report.notCovered.insert(thunk, addr);
301 }
302 return true;
303}
