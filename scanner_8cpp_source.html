27t_scan_status pesieve::ProcessScanner::scanForHollows(HANDLE processHandle, ModuleData& modData, RemoteModuleData &remoteModData, ProcessScanReport& process_report)
28{
29 BOOL isWow64 = FALSE;
30#ifdef _WIN64
31 is_process_wow64(processHandle, &isWow64);
32#endif
33 HeadersScanner hollows(processHandle, modData, remoteModData);
34 HeadersScanReport *scan_report = hollows.scanRemote();
35 if (!scan_report) {
36 process_report.appendReport(new UnreachableModuleReport(processHandle, modData.moduleHandle, modData.original_size, modData.szModName));
37 return SCAN_ERROR;
38 }
39
40 if (scan_report->archMismatch && isWow64) {
41#ifdef _DEBUG
42 std::cout << "Arch mismatch, reloading..." << std::endl;
43#endif
44 if (modData.reloadWow64()) {
45 delete scan_report; // delete previous report
46 scan_report = hollows.scanRemote();
47 }
48 }
49 scan_report->moduleFile = modData.szModName;
50 scan_report->isInPEB = modData.isModuleInPEBList();
51
52 t_scan_status is_suspicious = ModuleScanReport::get_scan_status(scan_report);
53 if (is_suspicious && !scan_report->isHdrReplaced()) {
54 is_suspicious = SCAN_NOT_SUSPICIOUS;
55 }
56 process_report.appendReport(scan_report);
57 return is_suspicious;

154ProcessScanReport* pesieve::ProcessScanner::scanRemote()
155{
156 this->isDEP = is_DEP_enabled(this->processHandle);
157
158 const bool is_64bit = pesieve::util::is_process_64bit(this->processHandle);
159
160 ProcessScanReport *pReport = new ProcessScanReport(this->args.pid, is_64bit);
161
162 char image_buf[MAX_PATH] = { 0 };
163 GetProcessImageFileNameA(this->processHandle, image_buf, MAX_PATH);
164 pReport->mainImagePath = device_path_to_win32_path(image_buf);
165
166 std::stringstream errorsStr;
167
168 // scan modules
169 size_t modulesScanned = 0;
170 size_t iatsScanned = 0;
171 try {
172 modulesScanned = scanModules(*pReport);
173 if (args.iat) {
174 iatsScanned = scanModulesIATs(*pReport);
175 }
176 } catch (std::exception &e) {
177 modulesScanned = 0;
178 iatsScanned = 0;
179 errorsStr << e.what();
180 }
181
182 // scan working set
183 size_t regionsScanned = 0;
184 try {
185 //dont't scan your own working set
186 if (peconv::get_process_id(this->processHandle) != GetCurrentProcessId()) {
187 regionsScanned = scanWorkingSet(*pReport);
188 }
189 } catch (std::exception &e) {
190 regionsScanned = 0;
191 errorsStr << e.what();
192 }
193
194 // throw error only if none of the scans was successful
195 if (!modulesScanned && !iatsScanned && !regionsScanned) {
196 throw std::runtime_error(errorsStr.str());
197 }
198 //post-process hooks
199 resolveHooksTargets(*pReport);
200
201 //post-process .NET modules
202 filterDotNetReport(*pReport);
203 return pReport;
204}

261size_t pesieve::ProcessScanner::scanModules(ProcessScanReport &pReport) //throws exceptions
262{
263 HMODULE hMods[1024] = { 0 };
264 const size_t modules_count = enum_modules(this->processHandle, hMods, sizeof(hMods), LIST_MODULES_ALL);
265 if (modules_count == 0) {
266 return 0;
267 }
268 if (args.imprec_mode != PE_IMPREC_NONE || args.iat != pesieve::PE_IATS_NONE) {
269 pReport.exportsMap = new peconv::ExportsMapper();
270 }
271
272 size_t counter = 0;
273 for (counter = 0; counter < modules_count; counter++) {
274 if (processHandle == nullptr) break;
275
276 //load module from file:
277 ModuleData modData(processHandle, hMods[counter], true);
278 ModuleScanReport *mappingScanReport = this->scanForMappingMismatch(modData, pReport);
279
280 //load the original file to make the comparisons:
281 if (!modData.loadOriginal()) {
282 if (!args.quiet) {
283 std::cout << "[!][" << args.pid << "] Suspicious: could not read the module file!" << std::endl;
284 }
285 //make a report that finding original module was not possible
286 pReport.appendReport(new UnreachableModuleReport(processHandle, hMods[counter], 0, modData.szModName));
287 continue;
288 }
289
290 // Don't scan modules that are in the ignore list
291 std::string plainName = peconv::get_file_name(modData.szModName);
292 if (is_in_list(plainName.c_str(), this->ignoredModules)) {
293 // ...but add such modules to the exports lookup:
294 if (pReport.exportsMap && modData.loadOriginal()) {
295 pReport.exportsMap->add_to_lookup(modData.szModName, (HMODULE)modData.original_module, (ULONGLONG)modData.moduleHandle);
296 }
297 if (!args.quiet) {
298 std::cout << "[*] Skipping ignored: " << std::hex << (ULONGLONG)modData.moduleHandle << " : " << modData.szModName << std::endl;
299 }
300 pReport.appendReport(new SkippedModuleReport(processHandle, modData.moduleHandle, modData.original_size, modData.szModName));
301 continue;
302 }
303 if (!args.quiet) {
304 std::cout << "[*] Scanning: " << modData.szModName << std::endl;
305 }
306 if (modData.isDotNet()) {
307 pReport.isManaged = true;
308#ifdef _DEBUG
309 std::cout << "[*] Skipping a .NET module: " << modData.szModName << std::endl;
310#endif
311 pReport.appendReport(new SkippedModuleReport(processHandle, modData.moduleHandle, modData.original_size, modData.szModName));
312 continue;
313 }
314 //load data about the remote module
315 RemoteModuleData remoteModData(processHandle, this->isReflection, hMods[counter]);
316 if (remoteModData.isInitialized() == false) {
317 //make a report that initializing remote module was not possible
318 pReport.appendReport(new MalformedHeaderReport(processHandle, hMods[counter], 0, modData.szModName));
319 continue;
320 }
321 t_scan_status is_hollowed = scanForHollows(processHandle, modData, remoteModData, pReport);
322 if (is_hollowed == SCAN_ERROR) {
323 continue;
324 }
325 if (is_hollowed == SCAN_NOT_SUSPICIOUS) {
326 //if the content does not differ, ignore the different name of the mapped file
327 mappingScanReport->status = SCAN_NOT_SUSPICIOUS;
328 }
329 if (pReport.exportsMap != nullptr) {
330 pReport.exportsMap->add_to_lookup(modData.szModName, (HMODULE) modData.original_module, (ULONGLONG) modData.moduleHandle);
331 }
332 // if hooks not disabled and process is not hollowed, check for hooks:
333 if (!args.no_hooks && (is_hollowed == SCAN_NOT_SUSPICIOUS)) {
334
335 const bool scan_data = ((this->args.data >= pesieve::PE_DATA_SCAN_ALWAYS) && (this->args.data != pesieve::PE_DATA_SCAN_INACCESSIBLE_ONLY))
336 || (!this->isDEP && (this->args.data == pesieve::PE_DATA_SCAN_NO_DEP));
337
338 const bool scan_inaccessible = (this->isReflection && (this->args.data >= PE_DATA_SCAN_INACCESSIBLE));
339
340 scanForHooks(processHandle, modData, remoteModData, pReport, scan_data, scan_inaccessible);
341 }
342 }
343 return counter;
344}
