166WorkingSetScanReport* pesieve::WorkingSetScanner::scanRemote()
167{
168 if (!memPage.isInfoFilled() && !memPage.fillInfo()) {
169#ifdef _DEBUG
170 std::cout << "[!] Could not fill: " << std::hex << memPage.start_va << " to: " << memPage.region_end << "\n";
171#endif
172 return nullptr;
173 }
174 // is the page executable?
175 const bool is_any_exec = isExecutable(memPage);
176 if (!is_any_exec) {
177 // probably not interesting
178 return nullptr;
179 }
180
181 if (memPage.mapping_type == MEM_MAPPED && memPage.isRealMapping()) {
182 //probably legit
183 return nullptr;
184 }
185
186 if (memPage.mapping_type == MEM_IMAGE) {
187 memPage.loadModuleName();
188 memPage.loadMappedName();
189 if (!isScannedAsModule(memPage)) {
190 scanImg();
191 }
192 const size_t region_size = (memPage.region_end) ? (memPage.region_end - memPage.region_start) : 0;
193 if (this->processReport.hasModuleContaining(memPage.region_start, region_size)) {
194 // the area was already scanned
195 return nullptr;
196 }
197 }
198#ifdef _DEBUG
199 std::cout << std::hex << memPage.start_va << ": Scanning executable area" << std::endl;
200#endif
201 WorkingSetScanReport* my_report = this->scanExecutableArea(memPage);
202 if (!my_report) {
203 return nullptr;
204 }
205 my_report->is_executable = true;
206 my_report->protection = memPage.protection;
207 my_report->mapping_type = memPage.mapping_type;
208 my_report->mapped_name = memPage.mapped_name;
209 return my_report;
210}
