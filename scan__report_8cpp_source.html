36bool pesieve::ProcessScanReport::hasAnyShownType(const pesieve::ProcessScanReport::t_report_filter &filter)

111size_t pesieve::ProcessScanReport::countHdrsReplaced() const
112{
113 size_t replaced = 0;
114 const t_report_type type = REPORT_HEADERS_SCAN;
115
116 std::set<ModuleScanReport*>::iterator itr;
117 for (itr = this->reportsByType[type].begin(); itr != this->reportsByType[type].end(); ++itr) {
118 ModuleScanReport* report = *itr;
119 if (ModuleScanReport::get_scan_status(report) == SCAN_SUSPICIOUS) {

121 if (!hdrRep) continue; //it should not happen

124 replaced++;
125 }
126 }
127 }

131pesieve::t_report pesieve::ProcessScanReport::generateSummary() const
132{
133 t_report summary = { 0 };
134 summary.pid = this->pid;
135 summary.is_64bit = this->is64bit;
136 summary.is_managed = this->isManaged;
137 summary.errors = static_cast<DWORD>(this->errorsCount);
138 summary.skipped = static_cast<DWORD>(this->reportsByType[REPORT_SKIPPED_SCAN].size());
139 summary.scanned = static_cast<DWORD>(this->reportsByType[REPORT_HEADERS_SCAN].size());
140
141 std::vector<ModuleScanReport*>::const_iterator itr = moduleReports.begin();
142 for (; itr != moduleReports.end(); ++itr) {
143 ModuleScanReport* report = *itr;
144 if (ModuleScanReport::get_scan_status(report) == SCAN_SUSPICIOUS) {
145 summary.suspicious++;
146 }
147 if (ModuleScanReport::get_scan_status(report) == SCAN_ERROR) {
148 summary.errors++;
149 }
150 }
151 summary.replaced = countHdrsReplaced();
152 summary.patched = countSuspiciousPerType(REPORT_CODE_SCAN);
153 summary.iat_hooked = countSuspiciousPerType(REPORT_IAT_SCAN);
154 summary.implanted_shc = countSuspiciousPerType(REPORT_MEMPAGE_SCAN);
155 summary.implanted_pe = countSuspiciousPerType(REPORT_ARTEFACT_SCAN);
156 summary.implanted = summary.implanted_shc + summary.implanted_pe;
157 summary.hdr_mod = countSuspiciousPerType(REPORT_HEADERS_SCAN) - summary.replaced;
158 summary.unreachable_file = countSuspiciousPerType(REPORT_UNREACHABLE_SCAN) + countResultsPerType(REPORT_UNREACHABLE_SCAN, pesieve::SCAN_ERROR);
159 summary.other = summary.suspicious - (summary.patched + summary.replaced + summary.implanted + summary.hdr_mod + summary.iat_hooked);
160 return summary;
161}

163std::string pesieve::ProcessScanReport::listModules(size_t level, const pesieve::ProcessScanReport::t_report_filter &filter, const t_json_level &jdetails) const
164{
165 std::stringstream stream;
166 //summary:
167 OUT_PADDED(stream, level, "\"scans\" : [\n");
168 bool is_first = true;
169 std::vector<ModuleScanReport*>::const_iterator itr;
170 for (itr = this->moduleReports.begin(); itr != this->moduleReports.end(); ++itr) {
171 ModuleScanReport *mod = *itr;
172 if (is_shown_type(mod->status, filter)) {
173 if (!is_first) {
174 stream << ",\n";
175 }
176 OUT_PADDED(stream, level + 1, "{\n");
177 mod->toJSON(stream, level + 2, jdetails);
178 stream << "\n";
179 OUT_PADDED(stream, level + 1, "}");
180 is_first = false;
181 }
182 }
183 if (moduleReports.size()) {
184 stream << "\n";
185 }
186 OUT_PADDED(stream, level, "]\n");
187 return stream.str();
