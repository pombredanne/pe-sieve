354void pesieve::ResultsDumper::makeAndJoinDirectories(std::stringstream& stream)
355{
356 if (!make_dump_dir(this->baseDir)) {
357 this->baseDir = ""; // reset path
358 }
359 std::string inner_dir = this->dumpDir;

362 }
363 if (!make_dump_dir(inner_dir)) {
364 this->dumpDir = ""; // reset path
365 }
366 if (baseDir.length() > 0) {
367 stream << baseDir;
