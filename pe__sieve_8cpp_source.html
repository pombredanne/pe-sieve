21 void check_access_denied(DWORD processID)
22 {
23 HANDLE hProcess = OpenProcess(PROCESS_QUERY_LIMITED_INFORMATION, FALSE, processID);
24 if (!hProcess) {
25 std::cerr << "-> Access denied. Try to run the scanner as Administrator." << std::endl;

33 case INTEGRITY_SYSTEM:
34 std::cerr << "-> Access denied. Could not access the system process." << std::endl;

40 hProcess = NULL;

58 HANDLE open_process(DWORD processID, bool reflection, bool quiet)
59 {
60 const DWORD basic_access = SYNCHRONIZE | PROCESS_VM_READ | PROCESS_QUERY_INFORMATION;
61 DWORD access = basic_access;
62 if (reflection) {
63 access |= pesieve::util::reflection_access | PROCESS_VM_OPERATION;
64 }
65
66 HANDLE hProcess = OpenProcess(access, FALSE, processID);
67
68 // if failed, try to open with the lower reflection access
69 if (!hProcess && access != basic_access && access != pesieve::util::reflection_access1) {
70 hProcess = OpenProcess(pesieve::util::reflection_access1, FALSE, processID);
71 }
72
73 // if failed, try to open with basic rights
74 if (!hProcess && access != basic_access) {
75 hProcess = OpenProcess( basic_access, FALSE, processID);
76 }
77
78 // check process compatibility
79 if (hProcess && !is_scanner_compatible(hProcess) && !quiet) {
80 util::print_in_color(WARNING_COLOR, "[!] Scanner mismatch! Try to use the 64bit version of the scanner!\n", true);
81 }
82
83 // opening succeeded, return the handle:
84 if (hProcess) {
85 return hProcess;
86 }
87
88 const DWORD last_err = GetLastError();
89
90 if (last_err == ERROR_ACCESS_DENIED) {
91 if (!quiet) {
92 std::cerr << "[-][" << processID << "] Could not open the process Error: " << last_err << std::endl;
93 //print more info:
94 check_access_denied(processID);
95 }
96
97 SetLastError(ERROR_ACCESS_DENIED);
98 throw std::runtime_error("Could not open the process: Access Denied");
99 return nullptr;
100 }
101 if (last_err == ERROR_INVALID_PARAMETER) {
102 if (!quiet) {
103 std::cerr << "-> Is this process still running?" << std::endl;
104 }
105 SetLastError(ERROR_INVALID_PARAMETER);
106 throw std::runtime_error("Could not open the process: Invalid Parameter");
107 }
108 return hProcess;
109 }

176pesieve::ReportEx* pesieve::scan_and_dump(IN const pesieve::t_params args)
177{
178 ReportEx *report = new ReportEx();
179
180 HANDLE orig_proc = nullptr; // original process handle
181 HANDLE cloned_proc = nullptr; // process reflection handle
182
183 if (!set_debug_privilege()) {
184 if (!args.quiet) std::cerr << "[-] Could not set debug privilege" << std::endl;
185 }
186
187 try {
188 orig_proc = open_process(args.pid, args.make_reflection, args.quiet);
189 HANDLE target_proc = orig_proc;
190
191 if (args.make_reflection) {
192 cloned_proc = make_process_reflection(orig_proc);
193 if (cloned_proc) {
194 target_proc = cloned_proc;
195 }
196 }
197
198 if (!args.quiet) {
199 if (cloned_proc) {
200 std::cout << "Using process reflection!\n";
201 }
202 else {
203 std::cout << "Using raw process!\n";
204 if (args.data >= PE_DATA_SCAN_INACCESSIBLE) {
205 print_in_color(WARNING_COLOR, "[WARNING] Scanning of inaccessible pages is possible only in reflection mode!\n");
206 }
207 }
208 }
209
210 const bool is_reflection = (cloned_proc) ? true : false;
211 ProcessScanner scanner(target_proc, is_reflection, args);
212 report->scan_report = scanner.scanRemote();
213
214 // dump process
215 if (report->scan_report) {
216 report->dump_report = make_dump(target_proc, is_reflection, args, *report->scan_report);
217 }
218 }
219 catch (std::exception &e) {
220 delete report;
221 report = nullptr;
222
223 if (!args.quiet) {
224 util::print_in_color(ERROR_COLOR, std::string("[ERROR] ") + e.what() + "\n", true);
225 }
226 }
227 if (cloned_proc) {
228 release_process_reflection(&cloned_proc);
229 }
230 CloseHandle(orig_proc);
231 return report;
232}
